signif <- 0.05

quadrant[m.cases >0 & m.local>0] <- 4  # high-high
quadrant[m.cases <0 & m.local<0] <- 1  # low-low    
quadrant[m.cases <0 & m.local>0] <- 2  # low-high
quadrant[m.cases >0 & m.local<0] <- 3  # high-low
